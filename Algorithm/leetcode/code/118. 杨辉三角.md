[118. 杨辉三角](https://leetcode-cn.com/problems/pascals-triangle/)

自己的垃圾代码
```cpp
class Solution {
public:
    vector<vector<int>> generate(int k) {
        vector<vector<int> > ans;
        if(k==0) return ans;
        else {ans.push_back({1});}
        for(int i=1;i<k;i++){
            vector<int> tmp(i+1,0);
            tmp[0]=1;tmp[i]=1;
            for(int j=1;j<i;j++)
                tmp[j]=ans[i-1][j-1]+ans[i-1][j];
            ans.push_back(tmp);
        }
        return ans;
    }
};
```
