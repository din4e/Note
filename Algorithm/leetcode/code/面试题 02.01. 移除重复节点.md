[面试题 02.01. 移除重复节点](https://leetcode-cn.com/problems/remove-duplicate-node-lcci/)

空间O(1) 这边可以省一个指针的变量；用p->next做判断；
```cpp
class Solution {
public:
    ListNode* removeDuplicateNodes(ListNode* head) {
        if(head==NULL) return head;
        ListNode*p=head;
        while(p!=NULL){
            ListNode *q=p->next,*qq=p;
            while(q!=NULL){
                if(q->val==p->val){
                    qq->next=q->next;
                }
                else qq=q;
                q=q->next;
            }
            p=p->next;
        }
        return head;
    }
};
```

```cpp
class Solution {
public:
    ListNode* removeDuplicateNodes(ListNode* head) {
        if(head==NULL) return head;
        ListNode *p=head->next,*pr=head;
        unordered_set<int> m;
        m.insert(head->val);
        while(p!=NULL){
            if(m.find(p->val)==m.end()){
                m.insert(p->val);pr=p;
            }
            else pr->next=p->next;
            p=p->next;
        }
        return head;
    }
};
```

最快的
```cpp
class Solution {
private:
bool vis[20001] = {0};
public:
    ListNode* removeDuplicateNodes(ListNode* head) {
        if(head == NULL || head->next == NULL) return head;

        ListNode* cur = head;
        ListNode* pre = head;
        vis[cur->val] = true;
        cur = cur->next;

        while(cur!=NULL)
        {
            if(vis[cur->val])
            {
                pre->next = cur->next;
                cur = pre->next;
            }
            else
            {
                vis[cur->val] = true;
                pre = cur;
                cur = cur->next;
            }
        }
        return head;
    }
};
```
